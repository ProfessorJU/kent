// script.js

document.addEventListener('DOMContentLoaded', function() {
    // Select all slide elements
    const slides = document.querySelectorAll('.slide');
    let currentSlide = 0;
    const slideInterval = 5000; // Time in milliseconds (5 seconds)

    // Ensure the first slide is visible on load
    slides[currentSlide].classList.add('active');

    function nextSlide() {
        // Hide the current slide
        slides[currentSlide].classList.remove('active');
        
        // Calculate the index of the next slide (cycles back to 0)
        currentSlide = (currentSlide + 1) % slides.length;
        
        // Show the new current slide
        slides[currentSlide].classList.add('active');
    }

    // Start the automatic slide show timer
    setInterval(nextSlide, slideInterval);
});
